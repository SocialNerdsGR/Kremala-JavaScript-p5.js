### For
```javascript
let myArray = ['Hello', 'how', 'are', 'you'];
for (let i = 0; i < myArray.length; i++) {
  console.log(myArray[i]);
}
```

### ForEach
```javascript
let myArray = ['Hello', 'how', 'are', 'you'];
myArray.forEach(element => {
  console.log(element);
});
```
